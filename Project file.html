#3 AR Model
import numpy as np
import matplotlib.pyplot as plt

# Step 1: Create synthetic data (sine wave + noise)
np.random.seed(0)
data = np.sin(np.linspace(0, 10, 100)) + np.random.normal(0, 0.2, 100)

# Step 2: Manual AR(1) prediction (next = prev * some weight)
predictions = [data[0]]
for t in range(1, len(data)):
    pred = 0.9 * data[t-1]  # using lag-1 with weight 0.9
    predictions.append(pred)

# Step 3: Plot
plt.plot(data, label='Original')
plt.plot(predictions, label='Predicted', color='orange')
plt.title("Simple Manual AR(1) Model")
plt.legend()
plt.show()

#6 Flow Map to show international student migration

import plotly.graph_objects as go

# Example flows: [from_lon, from_lat, to_lon, to_lat]
flows = [
    [78, 21, -95, 37],   # India to USA
    [104, 36, -95, 37],  # China to USA
    [10, 51, -3, 55],    # Germany to UK
]

fig = go.Figure()

for flow in flows:
    fig.add_trace(go.Scattergeo(
        lon=[flow[0], flow[2]],
        lat=[flow[1], flow[3]],
        mode='lines',
        line=dict(width=6, color='olive')
    ))

fig.update_layout(
    geo=dict(projection_type="natural earth"),
    title="Student Migration"
)

fig.show()
